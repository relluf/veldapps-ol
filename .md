	function dms(rad, s) {
	    /*- degrees,minutes, seconds */
	    if (rad < 0) {
	        s = s.charAt(1);
	        rad = -rad;
	    } else {
	        s = s.charAt(0);
	    }
	    var deg = parseInt(rad, 10);
	    rad -= deg;
	    rad *= 60;
	    var min = parseInt(rad, 10);
	    rad -= min;
	    rad *= 60;
	    if (rad < 0.001) {
	        return String.format("%d&deg;%d'\" %s", deg, min, s);
	    }
	    return String.format("%d&deg;%d'%.2f\" %s", deg, min, rad, s);
	}
